## 📄 core/tools.py — Tool Registry & Execution

```python
"""
Tool System — Static tools + self-generated dynamic tools.
"""

import importlib
import importlib.util
import inspect
import os
import sys
import subprocess
import json
import textwrap
from typing import Dict, List, Callable, Any, Optional
from dataclasses import dataclass, field
from pathlib import Path


@dataclass
class ToolDefinition:
    """Metadata for a tool that JARVIS can call."""
    name: str
    description: str
    parameters: Dict  # JSON schema for parameters
    function: Callable = None
    category: str = "utility"
    auto_approve: bool = False
    is_dynamic: bool = False  # True if self-generated
    source_file: str = None   # Path to generated code file


class ToolRegistry:
    """Registry of all available tools."""
    
    def __init__(self, settings):
        self.tools: Dict[str, ToolDefinition] = {}
        self.custom_tools_path = Path(settings.get("custom_tools_dir", "./custom_tools"))
        self.auto_approve_list = settings.get("auto_approve_tools", [])
        
        # Ensure custom_tools directory exists
        self.custom_tools_path.mkdir(parents=True, exist_ok=True)
        
        # Register built-in tools
        self._register_builtins()
        
        # Load any existing custom tools
        self._load_custom_tools()
    
    def register(self, tool: ToolDefinition):
        """Register a tool in the registry."""
        self.tools[tool.name] = tool
    
    def get(self, name: str) -> Optional[ToolDefinition]:
        return self.tools.get(name)
    
    def get_definitions(self) -> List[Dict]:
        """Return tool definitions in OpenAI function-calling format."""
        definitions = []
        for tool in self.tools.values():
            definitions.append({
                "type": "function",
                "function": {
                    "name": tool.name,
                    "description": tool.description,
                    "parameters": {
                        "type": "object",
                        "properties": tool.parameters,
                        "required": [
                            k for k, v in tool.parameters.items()
                            if v.get("required", False)
                        ],
                    },
                },
            })
        return definitions
    
    def execute(self, tool_name: str, arguments: Dict) -> str:
        """Execute a tool and return the result as a string."""
        tool = self.get(tool_name)
        if not tool:
            return f"Error: Tool '{tool_name}' not found."
        
        # Check auto-approve
        if not tool.auto_approve and tool_name not in self.auto_approve_list:
            # In practice, this would prompt the user for confirmation
            print(f"⚠️ Tool '{tool_name}' requires approval.")
            print(f"   Args: {arguments}")
            # Wait for user confirmation (Y/N)
        
        try:
            result = tool.function(**arguments)
            return str(result)
        except Exception as e:
            return f"Error executing '{tool_name}': {e}"
    
    def generate_tool(self, name: str, description: str, 
                      requirements: str, code: str = None) -> ToolDefinition:
        """
        Generate and register a new tool from LLM-generated code.
        This is JARVIS's self-evolution capability.
        """
        filepath = self.custom_tools_path / f"{name}.py"
        
        if code is None:
            # In practice, code would be generated by the LLM
            # based on the description and requirements
            raise ValueError("Code generation requires LLM input")
        
        # Write the tool code
        filepath.write_text(code)
        
        # Dynamically import the tool
        spec = importlib.util.spec_from_file_location(name, str(filepath))
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)
        
        # Find the main function (assumed to be `run` or function with tool name)
        tool_func = None
        for attr_name in dir(module):
            if attr_name == name or attr_name == "run":
                tool_func = getattr(module, attr_name)
                if callable(tool_func):
                    break
        
        if not tool_func:
            # Take the first callable function
            for attr_name in dir(module):
                attr = getattr(module, attr_name)
                if callable(attr) and not attr_name.startswith("_"):
                    tool_func = attr
                    break
        
        if not tool_func:
            raise ValueError(f"No callable function found in generated tool '{name}'")
        
        # Extract parameter info from function signature
        sig = inspect.signature(tool_func)
        parameters = {}
        for param_name, param in sig.parameters.items():
            param_type = "string"
            if param.annotation != inspect.Parameter.empty:
                if param.annotation == int:
                    param_type = "integer"
                elif param.annotation == float:
                    param_type = "number"
                elif param.annotation == bool:
                    param_type = "boolean"
            parameters[param_name] = {
                "type": param_type,
                "description": f"Parameter: {param_name}",
                "required": param.default == inspect.Parameter.empty,
            }
        
        tool_def = ToolDefinition(
            name=name,
            description=description,
            parameters=parameters,
            function=tool_func,
            category="dynamic",
            auto_approve=False,
            is_dynamic=True,
            source_file=str(filepath),
        )
        
        self.register(tool_def)
        return tool_def
    
    def _register_builtins(self):
        """Register all built-in tools."""
        builtins = [
            ToolDefinition(
                name="get_system_stats",
                description="Get CPU, RAM, disk, and network usage statistics",
                parameters={},
                function=self._builtin_system_stats,
                category="system",
                auto_approve=True,
            ),
            ToolDefinition(
                name="read_file",
                description="Read contents of a file",
                parameters={
                    "path": {
                        "type": "string",
                        "description": "Path to the file",
                        "required": True,
                    }
                },
                function=self._builtin_read_file,
                category="filesystem",
                auto_approve=False,
            ),
            ToolDefinition(
                name="write_file",
                description="Write content to a file",
                parameters={
                    "path": {
                        "type": "string",
                        "description": "Path to the file",
                        "required": True,
                    },
                    "content": {
                        "type": "string",
                        "description": "Content to write",
                        "required": True,
                    },
                },
                function=self._builtin_write_file,
                category="filesystem",
                auto_approve=False,
            ),
            ToolDefinition(
                name="execute_python",
                description="Execute Python code in a sandbox",
                parameters={
                    "code": {
                        "type": "string",
                        "description": "Python code to execute",
                        "required": True,
                    }
                },
                function=self._builtin_execute_python,
                category="code",
                auto_approve=False,
            ),
            ToolDefinition(
                name="get_weather",
                description="Get current weather for a location",
                parameters={
                    "location": {
                        "type": "string",
                        "description": "City name or coordinates",
                        "required": True,
                    }
                },
                function=self._builtin_weather,
                category="utility",
                auto_approve=True,
            ),
            ToolDefinition(
                name="web_search",
                description="Search the web for information",
                parameters={
                    "query": {
                        "type": "string",
                        "description": "Search query",
                        "required": True,
                    }
                },
                function=self._builtin_web_search,
                category="web",
                auto_approve=True,
            ),
            ToolDefinition(
                name="take_screenshot",
                description="Capture a screenshot of the screen",
                parameters={},
                function=self._builtin_screenshot,
                category="system",
                auto_approve=False,
            ),
            ToolDefinition(
                name="set_reminder",
                description="Set a timed reminder",
                parameters={
                    "message": {
                        "type": "string",
                        "description": "Reminder message",
                        "required": True,
                    },
                    "minutes": {
                        "type": "integer",
                        "description": "Minutes from now",
                        "required": True,
                    },
                },
                function=self._builtin_reminder,
                category="utility",
                auto_approve=True,
            ),
        ]
        
        for tool in builtins:
            self.register(tool)
    
    def _builtin_system_stats(self) -> str:
        import psutil
        cpu = psutil.cpu_percent(interval=0.5)
        ram = psutil.virtual_memory()
        disk = psutil.disk_usage("/")
        return (
            f"CPU: {cpu}% | "
            f"RAM: {ram.percent}% ({ram.used//1024**3}GB/{ram.total//1024**3}GB) | "
            f"Disk: {disk.percent}% ({disk.used//1024**3}GB/{disk.total//1024**3}GB)"
        )
    
    def _builtin_read_file(self, path: str) -> str:
        path = Path(path).resolve()
        if not path.exists():
            return f"File not found: {path}"
        return path.read_text(encoding="utf-8")
    
    def _builtin_write_file(self, path: str, content: str) -> str:
        path = Path(path).resolve()
        path.parent.mkdir(parents=True, exist_ok=True)
        path.write_text(content, encoding="utf-8")
        return f"Written {len(content)} bytes to {path}"
    
    def _builtin_execute_python(self, code: str) -> str:
        """Execute Python in a sandboxed subprocess."""
        # Security: runs in subprocess with timeout
        try:
            result = subprocess.run(
                [sys.executable, "-c", code],
                capture_output=True,
                text=True,
                timeout=30,
                env={},  # Isolated environment
            )
            output = result.stdout
            if result.stderr:
                output += f"\nSTDERR:\n{result.stderr}"
            return output or "(no output)"
        except subprocess.TimeoutExpired:
            return "Execution timed out (30s)"
        except Exception as e:
            return f"Execution error: {e}"
    
    def _builtin_weather(self, location: str) -> str:
        """Simple weather via wttr.in."""
        import requests
        try:
            resp = requests.get(f"https://wttr.in/{location}?format=%C+%t+%w+%h", timeout=10)
            return f"Weather in {location}: {resp.text}"
        except Exception as e:
            return f"Could not get weather: {e}"
    
    def _builtin_web_search(self, query: str) -> str:
        """Simple web search using DuckDuckGo."""
        try:
            import urllib.parse
            import requests
            from bs4 import BeautifulSoup
            
            url = f"https://html.duckduckgo.com/html/?q={urllib.parse.quote(query)}"
            resp = requests.get(url, headers={"User-Agent": "Mozilla/5.0"}, timeout=10)
            soup = BeautifulSoup(resp.text, "html.parser")
            
            results = []
            for result in soup.select(".result__body")[:5]:
                title_el = result.select_one(".result__title")
                snippet_el = result.select_one(".result__snippet")
                if title_el and snippet_el:
                    results.append(f"- {title_el.get_text(strip=True)}: {snippet_el.get_text(strip=True)}")
            
            return "\n".join(results) if results else "No results found."
        except Exception as e:
            return f"Search error: {e}"
    
    def _builtin_screenshot(self) -> str:
        """Take a screenshot and save it."""
        try:
            import pyautogui
            screenshot = pyautogui.screenshot()
            path = f"./logs/screenshots/screenshot_{int(__import__('time').time())}.png"
            screenshot.save(path)
            return f"Screenshot saved to {path}"
        except Exception as e:
            return f"Screenshot failed: {e}"
    
    def _builtin_reminder(self, message: str, minutes: int) -> str:
        """Set a reminder (in-memory for now)."""
        import threading
        def _notify():
            print(f"\n⏰ REMINDER: {message}")
        timer = threading.Timer(minutes * 60, _notify)
        timer.daemon = True
        timer.start()
        return f"Reminder set for {minutes} minutes: '{message}'"
    
    def _load_custom_tools(self):
        """Load any existing custom tools from the custom_tools directory."""
        if not self.custom_tools_path.exists():
            return
        
        for filepath in self.custom_tools_path.glob("*.py"):
            if filepath.name == "__init__.py":
                continue
            
            name = filepath.stem
            
            # Already registered? skip
            if name in self.tools:
                continue
            
            try:
                spec = importlib.util.spec_from_file_location(name, str(filepath))
                module = importlib.util.module_from_spec(spec)
                spec.loader.exec_module(module)
                
                # Find the main function
                tool_func = None
                for attr_name in dir(module):
                    if attr_name == name or attr_name == "run":
                        tool_func = getattr(module, attr_name)
                        if callable(tool_func):
                            break
                
                if tool_func:
                    sig = inspect.signature(tool_func)
                    parameters = {}
                    for param_name, param in sig.parameters.items():
                        parameters[param_name] = {
                            "type": "string",
                            "description": f"Parameter: {param_name}",
                            "required": param.default == inspect.Parameter.empty,
                        }
                    
                    self.register(ToolDefinition(
                        name=name,
                        description=f"Custom tool: {name}",
                        parameters=parameters,
                        function=tool_func,
                        category="dynamic",
                        is_dynamic=True,
                        source_file=str(filepath),
                    ))
                    print(f"🔧 Loaded custom tool: {name}")
            except Exception as e:
                print(f"⚠️ Failed to load custom tool '{filepath.name}': {e}")
```

### self-improvement Workflow

When a tool fails during execution:

1. JARVIS detects the error
2. JARVIS asks the LLM to analyze the error
3. LLM generates a fixed version of the code
4. Fixed code is written to the same file
5. `importlib.reload()` refreshes the module
6. Tool is re-executed

```python
def improve_tool(self, tool_name: str, error: str, 
                 fix_code: str) -> ToolDefinition:
    """Re-generate a tool with a fix applied."""
    tool = self.get(tool_name)
    if not tool or not tool.is_dynamic:
        raise ValueError(f"Cannot improve non-dynamic tool: {tool_name}")
    
    filepath = Path(tool.source_file)
    filepath.write_text(fix_code)
    
    # Reload the module
    spec = importlib.util.spec_from_file_location(tool_name, str(filepath))
    module = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(module)
    
    # ... re-extract the function and re-register
```
